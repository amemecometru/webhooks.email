Quick start

Go from zero to
receiving webhooks in 60 seconds.

curl Node.js Python 
# Create an endpoint (returns your webhook URL)
curl -X POST https://api.webhooks.email/v1/endpoints \
  -H "Authorization: Bearer $TOKEN"

# Send a test webhook
curl -X POST https://you.webhooks.email/hook \
  -H "Content-Type: application/json" \
  -d '{"test": true}'

# Forward to your server
curl -X PATCH https://api.webhooks.email/v1/endpoints/:id \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"destinations": ["https://myapp.com/webhooks"]}'
